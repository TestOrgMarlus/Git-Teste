mkdir 'pasta'
	criar nova pasta

cd 'pasta'
	entra em pasta

cd ..
	voltar à pasta anterior

git init
	inicializar o git na pasta atual

ls
	listar os arquivos do diretório





git status 
	status dos arquivos

vim Readme.md
	editar ou criar um arquivo .md

git add arquivo.arc
	trackear a última modificação do arquivo

git commit -m "Comentário sobre a modificação"
	enviar a modificação trackeada no master
	a modificação precisa ser adicionada antes de ser comitada




git log
	histórico de commits do arquivo

git log --decorate
	Log mais detalhado

git log --author="nome"
	log específico de um usuário que contenha a string "nome"

git shortlog
	resumo da quantidade e comentários dos commits de cada pessoa no projeto

git shortlog -sn
	mostrar só a quantidade de commits por pessoa

git log --graph
	mostrar os commit de maneira gráfica

git show exemplo
	botar a hash de um commit no lugar de exemplo para ver mais detalhadamente
	o que foi feito no commit específico.

	
git diff
	mostrar todas as mudanças feitas no diretorio antes do commit
	USE ANTES DE FAZER UM COMMIT!
	Mesma coisa de um git show, mas antes do commitgit

git diff --name-only
	mostra só o nome dos arquivos modificados

git commit -am "Comentário"
	adciona e commit no mesmo comando





git checkout arquivo.arc
	volta à versão anterior o arquivo no último commit
	ANTES DE ADICIoNAR

git reset HEAD arquivo.arc
	retira do tracking - revert o gid add

git reset --soft exemplo
	reverte o último commit, mas deixa a versão em staged/adicionada pra refazer o commit

git reset --mixed exemplo
	revert o arquivo pra modified, mas não staged

git reset --hard exemplo
	reverte o commit totalmente
	PARA TODOS OS RESET
		trocar exemplo pela hash da log anterior à qual você quer reverter



git remote add origin git@github.com:username/nomedapasta
	adiciona um repositório remoto do github no repositório local
	nome: origin - pode mudar pra qualquer outra coisa

git remote
	mostra o repositório remoto

git remote -v
	mostra o endereço do diretório remoto ligado ao local

git push -u origin master
	o push envia todas as modificações ao repositório origin/repositório remoto ligado
	-u: trackeia o comando para nas próximas só precisar usar git push
	origin: repositório remoto para o qual vão as modificações
	master: repositório mestre do qual saem as modificações


	




ctrl+insert
	colar o texto selecionado
